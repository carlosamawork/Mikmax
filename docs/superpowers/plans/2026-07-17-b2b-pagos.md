```ts
export interface PaymentSpec {
  transfer: boolean;
  giro: boolean;
}

export function parsePaymentSpec(json: string | null | undefined): PaymentSpec | null {
  if (!json) return null;
  let raw: unknown;
  try {
    raw = JSON.parse(json);
  } catch {
    return null;
  }
  if (typeof raw !== 'object' || raw === null) return null;
  const o = raw as Record<string, unknown>;
  if (typeof o.transfer !== 'boolean' || typeof o.giro !== 'boolean') return null;
  return {transfer: o.transfer, giro: o.giro};
}

// 'transfer' | 'giro' si el metodo es un manual gestionado; null -> no tocar jamas.
function classify(name?: string | null): 'transfer' | 'giro' | null {
  const n = (name ?? '').toLowerCase();
  if (n.includes('transferencia')) return 'transfer';
  if (n.includes('giro')) return 'giro';
  return null;
}

export function hiddenPaymentMethodIds(
  validated: boolean,
  spec: PaymentSpec | null,
  methods: {id: string; name?: string | null}[],
): string[] {
  const hidden: string[] = [];
  for (const m of methods) {
    const kind = classify(m.name);
    if (!kind) continue;
    const granted = validated && spec !== null && spec[kind];
    if (!granted) hidden.push(m.id);
  }
  return hidden;
}
```
