**Goal:** Tarifas de envío B2B propias (visibles solo para clientes validados) + override de envío por cliente, espejo del patrón del descuento.

**Architecture:** Las tarifas B2B viven en Ajustes→Envíos con prefijo `B2B ` (default). Una Delivery Customization Function nueva oculta a cada tipo de cliente las tarifas ajenas (con salvaguarda por grupo). El override por cliente (`b2b.shipping`, `{type:'percent'|'price'}`) lo aplica el target de envío de la Function de descuentos existente, siempre como porcentaje calculado por opción.

**Tech Stack:** Shopify Functions (TS, api 2026-04), app embebida Preact, vitest.

**Spec:** `docs/superpowers/specs/2026-07-17-b2b-envios-design.md`

```ts
export type ShippingSpec = {type: 'percent'; percent: number} | {type: 'price'; amount: number};

export function parseShippingSpec(json: string | null | undefined): ShippingSpec | null {
  if (!json) return null;
  let raw: unknown;
  try {
    raw = JSON.parse(json);
  } catch {
    return null;
  }
  if (typeof raw !== 'object' || raw === null) return null;
  const o = raw as Record<string, unknown>;
  if (o.type === 'percent') {
    return typeof o.percent === 'number' ? {type: 'percent', percent: o.percent} : null;
  }
  if (o.type === 'price') {
    return typeof o.amount === 'number' ? {type: 'price', amount: o.amount} : null;
  }
  return null;
}

const clamp = (n: number) => Math.max(0, Math.min(n, 100));

// % de descuento a aplicar a UNA opcion de envio segun su coste. Nunca sube precios.
export function shippingDiscountPercent(spec: ShippingSpec | null, cost: number): number {
  if (!spec) return 0;
  if (spec.type === 'percent') return clamp(spec.percent);
  if (cost <= 0 || spec.amount >= cost) return 0;
  return clamp(((cost - spec.amount) / cost) * 100);
}

export function shippingMessage(locale: string | null | undefined): string {
  return locale === 'es' ? 'Envío profesional' : 'Professional shipping';
}
```

```ts
import {
  DeliveryDiscountSelectionStrategy,
  DiscountClass,
  DeliveryInput,
  CartDeliveryOptionsDiscountsGenerateRunResult,
} from '../generated/api';
import {parseShippingSpec, shippingDiscountPercent, shippingMessage} from './shipping';

export function cartDeliveryOptionsDiscountsGenerateRun(
  input: DeliveryInput,
): CartDeliveryOptionsDiscountsGenerateRunResult {
  if (!input.discount.discountClasses.includes(DiscountClass.Shipping)) {
    return {operations: []};
  }
  const customer = input.cart.buyerIdentity?.customer;
  if (customer?.validated?.value !== 'true') {
    return {operations: []};
  }
  const spec = parseShippingSpec(customer?.shipping?.value);
  if (!spec) {
    return {operations: []};
  }

  const message = shippingMessage(input.cart.locale?.value);
  const candidates = [];
  for (const group of input.cart.deliveryGroups) {
    for (const option of group.deliveryOptions) {
      const percent = shippingDiscountPercent(spec, Number(option.cost.amount));
      if (percent <= 0) continue;
      candidates.push({
        message,
        targets: [{deliveryOption: {handle: option.handle}}],
        value: {percentage: {value: percent}},
      });
    }
  }
  if (!candidates.length) {
    return {operations: []};
  }
  return {
    operations: [
      {
        deliveryDiscountsAdd: {
          candidates,
          selectionStrategy: DeliveryDiscountSelectionStrategy.All,
        },
      },
    ],
  };
}
```

```ts
const PREFIX = 'B2B ';

export function hiddenOptionHandles(
  validated: boolean,
  groups: {deliveryOptions: {handle: string; title?: string | null}[]}[],
): string[] {
  const hidden: string[] = [];
  for (const g of groups) {
    const isB2b = (t?: string | null) => (t ?? '').startsWith(PREFIX);
    const hasB2b = g.deliveryOptions.some((o) => isB2b(o.title));
    for (const o of g.deliveryOptions) {
      if (validated) {
        // Salvaguarda: sin tarifas B2B en el grupo, el B2B ve las publicas.
        if (hasB2b && !isB2b(o.title)) hidden.push(o.handle);
      } else {
        if (isB2b(o.title)) hidden.push(o.handle);
      }
    }
  }
  return hidden;
}
```
