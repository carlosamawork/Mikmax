# i18n EN/ES Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add English (base, root) + Spanish (`/es/`) bilingual support to the site, gated by a feature flag that defaults to English-only behavior.

**Architecture:** A root `middleware.ts` strips the `/es` prefix to an internal rewrite and sets an `x-locale` request header; a server `getLocale()` reads it. Three translation surfaces — Sanity (field-level internationalized arrays + GROQ fallback), Shopify (`@inContext`), and a hand-rolled UI dictionary — each resolve to English when the flag is off or a translation is missing.

**Tech Stack:** Next.js 15 App Router, Sanity v3 (`@sanity/internationalized-array`), Shopify Storefront API (`@inContext`), Vitest, SCSS.

## Global Constraints

- Prettier: **no semicolons**, single quotes, 100 char width, **no bracket spacing** (`{a}` not `{ a }`). Copy this style in every code block.
- TypeScript strict mode; `npm run typecheck` must pass.
- Feature flag env var name is exactly **`NEXT_PUBLIC_I18N_ES`**; enabled only when its value is the string `'true'`.
- Locale codes are exactly `'en'` and `'es'`. `'en'` is the default and source of truth.
- Tests live in `__tests__/**/*.test.ts`, run with Vitest (`node` environment, `@` alias → repo root).
- Sanity queries MUST keep the `{next: {tags, revalidate}}` third argument (revalidation contract in CLAUDE.md).
- Never call `lib/shopify.js` from Client Components; never put cart logic outside `context/shopContext.js`.
- Commit after each task. Do not push or merge without asking the user.
- Flag OFF must reproduce today's behavior exactly (English-only, no `/es` routes, no `hreflang`).

// lib/i18n/localizedHref.ts
import type {Locale} from './config'

export function localizedHref(path: string, locale: Locale): string {
  if (locale === 'en') return path
  if (path === '/') return '/es'
  if (path === '/es' || path.startsWith('/es/')) return path
  return `/es${path.startsWith('/') ? path : `/${path}`}`
}

// middleware.ts
import {NextRequest, NextResponse} from 'next/server'
import {DEFAULT_LOCALE, isI18nEnabled} from '@/lib/i18n/config'

function setLocaleHeader(req: NextRequest, locale: 'en' | 'es') {
  const requestHeaders = new Headers(req.headers)
  requestHeaders.set('x-locale', locale)
  return NextResponse.next({request: {headers: requestHeaders}})
}

function prefersSpanish(req: NextRequest): boolean {
  const al = req.headers.get('accept-language') ?? ''
  return al.toLowerCase().split(',')[0]?.trim().startsWith('es') ?? false
}

export function middleware(req: NextRequest) {
  const {pathname} = req.nextUrl
  const isEs = pathname === '/es' || pathname.startsWith('/es/')

  if (!isI18nEnabled()) {
    if (isEs) {
      const url = req.nextUrl.clone()
      url.pathname = pathname.replace(/^\/es/, '') || '/'
      return NextResponse.redirect(url)
    }
    return setLocaleHeader(req, DEFAULT_LOCALE)
  }

  if (isEs) {
    const url = req.nextUrl.clone()
    url.pathname = pathname.replace(/^\/es/, '') || '/'
    const res = NextResponse.rewrite(url, {request: {headers: withLocale(req, 'es')}})
    return res
  }

  const cookie = req.cookies.get('NEXT_LOCALE')?.value
  if (!cookie && prefersSpanish(req)) {
    const url = req.nextUrl.clone()
    url.pathname = `/es${pathname === '/' ? '' : pathname}`
    return NextResponse.redirect(url, 308)
  }

  return setLocaleHeader(req, 'en')
}

function withLocale(req: NextRequest, locale: 'en' | 'es') {
  const h = new Headers(req.headers)
  h.set('x-locale', locale)
  return h
}

export const config = {
  matcher: ['/((?!admin|api|_next|.*\\..*).*)'],
}
```

// components/Layout/LanguageSwitcher/LanguageSwitcher.tsx
'use client'

import {usePathname, useRouter} from 'next/navigation'
import {setCookie} from 'cookies-next'
import type {Locale} from '@/lib/i18n/config'
import {localizedHref} from '@/lib/i18n/localizedHref'
import styles from './LanguageSwitcher.module.scss'

export default function LanguageSwitcher({current}: {current: Locale}) {
  const pathname = usePathname()
  const router = useRouter()

  function switchTo(locale: Locale) {
    if (locale === current) return
    setCookie('NEXT_LOCALE', locale, {maxAge: 60 * 60 * 24 * 365, path: '/'})
    const base = pathname.replace(/^\/es(\/|$)/, '/')
    router.push(localizedHref(base, locale))
  }

  return (
    <div className={styles.switcher}>
      <button aria-pressed={current === 'en'} onClick={() => switchTo('en')}>
        EN
      </button>
      <button aria-pressed={current === 'es'} onClick={() => switchTo('es')}>
        ES
      </button>
    </div>
  )
}
```

### Task 11: Migration script (wrap existing values into en)

**Files:**
- Create: `scripts/migrate-i18n.ts`

**Interfaces:**
- A standalone node script using `@sanity/client` with a write token (`SANITY_WRITE_TOKEN`) that, per document type and field, converts a scalar value `X` into `[{_key:'en', value: X}]` only when the field is not already an array of language objects (idempotent).

- [ ] **Step 1: Write the script**

```ts
// scripts/migrate-i18n.ts
import {createClient} from '@sanity/client'

const client = createClient({
  projectId: process.env.NEXT_PUBLIC_SANITY_PROJECT_ID!,
  dataset: process.env.NEXT_PUBLIC_SANITY_DATASET!,
  apiVersion: process.env.NEXT_PUBLIC_SANITY_API_VERSION!,
  token: process.env.SANITY_WRITE_TOKEN!,
  useCdn: false,
})

// Map of _type -> string fields to wrap. Extend as schemas are converted.
const TARGETS: Record<string, string[]> = {
  settings: ['siteTitle', 'siteDescription'],
  // home, legal, page, post, look, set, mikmaxForBusiness, b2bArea: fill per Task 12
}

function alreadyLocalized(v: unknown): boolean {
  return Array.isArray(v) && v.every((e) => e && typeof e === 'object' && '_key' in e)
}

async function run() {
  for (const [type, fields] of Object.entries(TARGETS)) {
    const docs = await client.fetch<Record<string, unknown>[]>(
      `*[_type == $type && !(_id in path('drafts.**'))]`,
      {type},
    )
    for (const doc of docs) {
      const patch: Record<string, unknown> = {}
      for (const f of fields) {
        const value = doc[f]
        if (value != null && !alreadyLocalized(value)) {
          patch[f] = [{_key: 'en', value}]
        }
      }
      if (Object.keys(patch).length) {
        await client.patch(doc._id as string).set(patch).commit()
        console.log(`migrated ${doc._id as string}: ${Object.keys(patch).join(', ')}`)
      }
    }
  }
  console.log('done')
}

run().catch((e) => {
  console.error(e)
  process.exit(1)
})
```

Run (user provides token/env): `npx tsx scripts/migrate-i18n.ts`
Expected: logs migrated doc ids; re-running logs nothing (idempotent). PortableText fields use the `localizedPortableText` shape `[{_key:'en', value: <blocks>}]` — handle them with a parallel `PORTABLE_TARGETS` map and the same wrap.

import {isI18nEnabled} from '@/lib/i18n/config'

export function localeAlternates(path: string) {
